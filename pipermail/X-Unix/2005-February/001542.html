here's two that rely on the whatismyip.com website:

#using lynx --dump:
url="whatismyip.com"
/sw/bin/lynx --nocolor --dump $url | head -10 |  grep "Your IP" |  awk 
'{print $4}'

# using curl:
curl -s "http://whatismyip.com" | grep "Your IP" | awk '{print $4}'
