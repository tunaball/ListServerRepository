Might try this:

#! /usr/bin/perl

use strict;
use warnings;

MAIN:
{
    opendir(DIR, ".") or die "error: opendir: $!\n";
    my @files = readdir(DIR);
    close(DIR);

    foreach my $f (@files) {
        my $inode = (stat($f))[1];

        print "$f: $inode\n";
    }

    exit 0;
}
