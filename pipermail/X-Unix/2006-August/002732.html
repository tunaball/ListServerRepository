The following example out to get you started:

   $ ls
   $ foo=`ls` ; if [[ "$foo" = "" ]] ; then echo "There are no files  
in this directory" ; else echo "These files are in this directory:  
$foo" ; fi
   There are no files in this directory
   $ > bar
    $ ls
   bar
   $ foo=`ls` ; if [[ "$foo" = "" ]] ; then echo "There are no files  
in this directory" ; else echo "These files are in this directory:  
$foo" ; fi
