Handy-dandy zombie factory (use at your own risk):

----CUT----OW----CUT----OW----CUT----
#include <sys/types.h>
#include <unistd.h>

#define ZOMBIES 2
int main() {
  int i;

  for(i = 0; i < ZOMBIES; i++) {
    if (fork() == 0) {
      /* child */
      exit(0);
    }
  }
  /* if you just exit here, init will reap the zombies */
  sleep(10000);
}
----CUT----OW----CUT----OW----CUT----
